function renderSnakeScreen(t1, t2) { ctx.clearRect(0,0, canvas.width, canvas.height); // Fetch the current theme's primary accent color const style = getComputedStyle(document.documentElement); const themeAccent = style.getPropertyValue('--theme-c1').trim(); ctx.fillStyle = themeAccent; ctx.textAlign = "center"; // Draw Main Text (S N A K E / GAME OVER) ctx.font = "bold 16px sans-serif"; ctx.fillText(t1, 120, 80); // Draw Sub Text (Click to Start / Refreshing...) ctx.font = "10px sans-serif"; ctx.fillText(t2, 120, 100); }
